<stmt> -> <assignmentStmt> | <declarativeStmt> | <conditionalStmt> SEMICOLON.
<type-specifier> -> KEYWORD_INT | KEYWORD_STRING | KEYWORD_BOOLEAN | KEYWORD_FLOAT .
<declarativeStmt> -> <type-specifier> <variable_list> SEMICOLON .
<variable_list> -> IDENTIFIER <more_ids> .
<more_ids> -> COMMA <variable_list> | EPSILON .
<conditionalStmt> -> KEYWORD_IF PARENTHESES_OPEN <booleanExpression> PARENTHESES_CLOSED <stmt> <otherStmts> <elseStmt> .
<elseStmt> -> KEYWORD_ELSE <stmt> <otherStmts> SEMICOLON |  EPSILON .
<otherStmts> -> <stmt> <otherStmts> | EPSILON .
<booleanExpression> -> PARENTHESES_OPEN <booleanExpression> PARENTHESES_CLOSED <logicalOperator> PARENTHESES_OPEN <booleanExpression> PARENTHESES_CLOSED | NOT PARENTHESES_OPEN <booleanExpression> PARENTHESES_CLOSED | <constrainedVars> <relationalOp> <constrainedVars> .
<logicalOperators> -> AND | OR .
<constrainedVars> -> IDENTIFIER | INT_NUM | FLOAT .
<relationalOp> -> LT | LE | EQ | GT | GE | NE .
<operator> -> ADD | SUB | MUL | DIV | MOD . 
<assignmentStmt> -> <lhs> ASSIGNOP <rhs> SEMICOLON .
<lhs> -> IDENTIFIER .
<rhs> -> <arithmeticExpression> .
<arithmeticExpression>-> <arithmeticExpression> <operator> <arithmeticExpression> .
<arithmeticExpression>-> PARENTHESES_OPEN <arithmeticExpression> PARENTHESES_CLOSED | <type-specifier> .
<var> -> IDENTIFIER | INT_NUM | FLOAT | STR .
