<stmt> -> <declarativeStmt> <stmt> | <assignmentStmt> <stmt> | <conditionalStmt> <stmt> | <recurStmt> <stmt> | EPSILON .
<assignmentStmt> -> IDENTIFIER ASSIGNOP <rhs> SEMICOLON .
<operator> -> ADD | SUB | DIV | MUL | MOD . 
<constrainedVars> -> IDENTIFIER | INT_NUM | FLOAT .
<rhs> -> <arithmetic_expression> | INT_NUM | FLOAT .
<arithmetic_expression> -> PARENTHESES_OPEN <constrainedVars> <operator> <constrainedVars> PARENTHESES_CLOSED .
<declarativeStmt> -> <type-specifier> <variable_list> SEMICOLON .
<type-specifier> -> KEYWORD_INT | KEYWORD_STRING | KEYWORD_FLOAT .
<variable_list> -> IDENTIFIER <more_ids> .
<more_ids> -> COMMA IDENTIFIER <more_ids> | EPSILON .
<conditionalStmt> -> KEYWORD_IF PARENTHESES_OPEN <booleanExpression> PARENTHESES_CLOSED PARENTHESES_OPEN <stmt> PARENTHESES_CLOSED <conditionalContd> . 
<conditionalContd> -> KEYWORD_ELSE PARENTHESES_OPEN <stmt> PARENTHESES_CLOSED | EPSILON .
<booleanExpression> -> <constrainedVars> <relationalOp> <constrainedVars> .
<relationalOp> -> LT | LE | EQ | GT | GE | NE .
<logicalOperators> -> AND | OR .
<recurStmt> -> KEYWORD_WHILE PARENTHESES_OPEN <booleanExpression> PARENTHESES_CLOSED PARENTHESES_OPEN <stmt> PARENTHESES_CLOSED .
